\section{Set Kitty as the Ubuntu default terminal}

\begin{minted}{bash}
sudo update-alternatives --install /usr/bin/x-terminal-emulator x-terminal-emulator `which kitty` 50
sudo update-alternatives --config x-terminal-emulator
\end{minted}
